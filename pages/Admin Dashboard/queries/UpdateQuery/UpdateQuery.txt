UPDATE pius.email_templates SET
    payment_type = COALESCE({{ update_form.formData.payment_type }}, payment_type),
    version = COALESCE({{ update_form.formData.version }}, version),
    is_active = COALESCE({{ update_form.formData.is_active }}, is_active),
    subject = COALESCE({{ update_form.formData.subject }}, subject),
    body = COALESCE({{ update_form.formData.body }}, body),
    updated_at = NOW()   -- auto timestamp update
WHERE template_id = {{ data_table.selectedRow.template_id }};

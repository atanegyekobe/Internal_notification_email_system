WITH rows_raw AS (
  SELECT *
  FROM json_to_recordset(
    '{{ JSON.stringify(appsmith.store.failedLogsToInsert || []) }}'
  ) AS (
      staff_ids TEXT,
      full_name TEXT,
      email TEXT,
      amount TEXT,            -- ← change NUMERIC → TEXT
      payment_type TEXT,
      template_id INT,
      message_hash TEXT
  )
),
rows_clean AS (
  SELECT
    staff_ids,
    full_name,
    email,
    -- Remove commas + cast to NUMERIC
    REPLACE(amount, ',', '')::NUMERIC AS amount,
    payment_type,
    template_id,
    message_hash
  FROM rows_raw
)
INSERT INTO pius.email_logs (
  batch_ref,
  job_id,
  recipient_id,
  recipient_email,
  recipient_name,
  payment_type,
  payment_amount,
  template_id,
  message_hash,
  status,
  created_by,
  created_at
)
SELECT
  '{{appsmith.store.currentFailedBatchRef}}',
	{{appsmith.store.currentFailedJobId}},
  staff_ids,
  email,
  full_name,
  payment_type,
  amount,             -- now clean numeric
  template_id,
  message_hash,
  'pending',
  '{{appsmith.store.currentUser.email}}',
  NOW()
FROM rows_clean;

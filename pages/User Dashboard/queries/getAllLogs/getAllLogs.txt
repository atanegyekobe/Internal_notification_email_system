SELECT 
  job_id,
  recipient_name,
  batch_ref,
  recipient_id,
  updated_at,
  status,
  recipient_email,
  payment_type,
  created_by,
  created_at,
  payment_amount,
  template_id,
  error_message,
  message_hash,
  updated_by,
	sent_at
FROM email_logs
WHERE created_by = '{{ appsmith.store.currentUser.email }}'
ORDER BY created_at DESC;
